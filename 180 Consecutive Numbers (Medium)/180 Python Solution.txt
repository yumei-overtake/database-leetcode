import pandas as pd

def consecutive_numbers(logs: pd.DataFrame) -> pd.DataFrame:
    logs = logs.sort_values(by='id',ascending=True)
    logs['next_num1']=logs['num'].shift(-1)
    logs['next_num2']=logs['num'].shift(-2)
    df = logs[(logs['num']==logs['next_num1']) & (logs['next_num1']==logs['next_num2'])][['num']].drop_duplicates()
    return df.rename(columns={'num':'ConsecutiveNums'})
