import pandas as pd

def market_analysis(users: pd.DataFrame, orders: pd.DataFrame, items: pd.DataFrame) -> pd.DataFrame:
    df = orders[orders['order_date'].dt.year==2019].groupby('buyer_id')['order_id'].count().reset_index()
    df_2 = users.merge(df,left_on='user_id',right_on='buyer_id',how='left')[['user_id','join_date','order_id']] \
        .rename(columns={'user_id':'buyer_id',"order_id":"orders_in_2019"})
    return df_2.fillna(0)
