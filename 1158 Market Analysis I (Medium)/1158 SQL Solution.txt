SELECT u.user_id as buyer_id,
       u.join_date,
       isnull(count(distinct o.order_id),0) as orders_in_2019 
FROM Users as u
Left join Orders as o 
         on o.buyer_id=u.user_id
         and year(o.order_date)=2019
GROUP BY u.user_id,
       u.join_date
