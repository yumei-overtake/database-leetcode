import pandas as pd
import numpy as np


def triangle_judgement(triangle: pd.DataFrame) -> pd.DataFrame:
# Define the "Triangle Inequality Theorem" condition
    is_triangle = (triangle['x']+triangle['y']>triangle['z'] ) & \
    (triangle['y']+triangle['z']>triangle['x'] ) & \
    (triangle['x']+triangle['z']>triangle['y'] ) 

# Apply np.where
    triangle['triangle']= np.where(is_triangle, 'Yes', 'No')

    return triangle
