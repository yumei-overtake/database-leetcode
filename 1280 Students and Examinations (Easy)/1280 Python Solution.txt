import pandas as pd

def students_and_examinations(students: pd.DataFrame, subjects: pd.DataFrame, examinations: pd.DataFrame) -> pd.DataFrame:
    df = students.merge(subjects,how='cross')
    df2 = examinations.groupby(['student_id','subject_name']).agg(attended_exams =('subject_name','count')).reset_index()
    df3= df.merge(df2,how='left',on=['student_id','subject_name'])[['student_id','student_name','subject_name','attended_exams']]
    df3['attended_exams']=df3['attended_exams'].fillna(0)
    return df3.sort_values(by=['student_id','subject_name'])
