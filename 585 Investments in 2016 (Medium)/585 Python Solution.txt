import pandas as pd

def find_investments(insurance: pd.DataFrame) -> pd.DataFrame:
    insurance['2015_cnt']=insurance.groupby('tiv_2015')['pid'].transform('count')
    insurance['location_cnt']=insurance.groupby(['lat','lon'])['pid'].transform('count')
    df = insurance[(insurance['2015_cnt']>1) & (insurance['location_cnt']==1)]
    return pd.DataFrame({'tiv_2016': [df['tiv_2016'].sum().round(2)]})
